CREATE TABLE sensorstate (
	fieldid ascii,
	nodeid ascii,
	sensorid ascii,
	sensortype ascii,
	latitude float,
	longitude float,
	lastcontacted timestamp,
	PRIMARY KEY (fieldid, nodeid, sensorid)
);

CREATE TABLE fieldstate (
	fieldid ascii,
	name ascii,
	users list<text>,
	PRIMARY KEY (fieldid)
);

CREATE TABLE configurationmessages (
	fieldid ascii,
	messageid uuid,
	payload ascii,
	logged int,
	PRIMARY KEY (fieldid, logged, messageid)
